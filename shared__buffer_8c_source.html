55 static void _shared_buffer_init_bitmap()
56 {
57  _shared_buffer_bitmap = _shared_buffer_zone.ptr;

60  bitmap = bitmap_wrap(_shared_buffer_bitmap, _shared_buffer_bitmap_len);

74 int shared_buffer_create(uint8_t** res_buffer, size_t size)
75 {
76  int buf_id = _shared_buffer_alloc_id();
77  if (buf_id < 0) {
78  return -ENOMEM;
79  }
80 
81  size_t act_size = size + sizeof(shared_buffer_header_t);
82 
83  int blocks_needed = (act_size + SHBUF_BLOCK_SIZE - 1) / SHBUF_BLOCK_SIZE;
84  int start = bitmap_find_space(bitmap, blocks_needed);
85  if (start < 0) {
86  return -ENOMEM;
87  }
88 
89  shared_buffer_header_t* space = (shared_buffer_header_t*)_shared_buffer_to_vaddr(start);
90  space->len = act_size;
91  bitmap_set_range(bitmap, start, blocks_needed);
92  vmm_tune_pages((uint32_t)space, act_size, PAGE_WRITABLE | PAGE_EXECUTABLE | PAGE_READABLE | PAGE_USER);
93 
94  *res_buffer = (uint8_t*)&space[1];
95  buffers[buf_id] = (uint8_t*)&space[1];
96 #ifdef SHARED_BUFFER_DEBUG
97  log("Buffer created at %x %d", buffers[buf_id], buf_id);
98 #endif
99  return buf_id;
100 }

103 {

117 int shared_buffer_free(int id)
118 {
119  if (unlikely(id < 0 || SHBUF_MAX_BUFFERS <= id)) {
120  return -EINVAL;
121  }
122  if (unlikely(buffers[id] == 0)) {
123  return -EINVAL;
124  }
125 
126  shared_buffer_header_t* sptr = (shared_buffer_header_t*)buffers[id];
127  size_t blocks_to_delete = (sptr[-1].len + SHBUF_BLOCK_SIZE - 1) / SHBUF_BLOCK_SIZE;
128  bitmap_unset_range(bitmap, _shared_buffer_to_index((uint32_t)&sptr[-1]), blocks_to_delete);
129  buffers[id] = 0;
