185 int tasking_exec(const char* path, const char** argv, const char** env)
186 {
187  thread_t* thread = RUNNIG_THREAD;
188  proc_t* p = RUNNIG_THREAD->process;
189  char* kpath = 0;
190  int kargc = 0;
191  char** kargv = 0;
192  char** kenv = 0;
193 
194  if (!str_validate_len(path, 128)) {
195  return -EINVAL;
196  }
197  kpath = kmem_bring_to_kernel(path, strlen(path) + 1);
198 
199  if (argv) {
200  if (!ptrarr_validate_len(argv, 128)) {
201  return -EINVAL;
202  }
203  kargc = ptrarr_len(argv);
204 
205  /* Validating arguments size */
206  uint32_t data_len = 0;
207  for (int argi = 0; argi < kargc; argi++) {
208  if (!str_validate_len(argv[argi], 128)) {
209  return -EINVAL;
210  }
211  data_len += strlen(argv[argi]) + 1;
212  if (data_len > 128) {
213  return -EINVAL;
214  }
215  }
216 
217  kargv = kmem_bring_to_kernel_ptrarr(argv, kargc);
218  }
219 
220  /* Cleaning proc */
221  dynamic_array_clear(&p->zones);
222  proc_kill_all_threads_except(p, thread);
223  p->main_thread = thread;
224  p->pid = thread->tid;
225  fpu_reset_state(p->main_thread->fpu_state);
226 
227  int ret = _tasking_do_exec(p, kpath, kargc, kargv, 0);
228 
229 #ifdef TASKING_DEBUG
230  if (!ret) {
231  log("Exec %s : pid %d", kpath, p->pid);
232  }
233 #endif
234 
235  kfree(kpath);
236  for (int argi = 0; argi < kargc; argi++) {
237  kfree(kargv[argi]);
238  }
239  kfree(kargv);
240 
241  return ret;
242 }
