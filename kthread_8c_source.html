22 int kthread_setup(proc_t* p)
23 {
24  p->pid = proc_alloc_pid();
25  p->pgid = p->pid;
26  p->is_kthread = true;
27  /* allocating kernel stack */
28  p->main_thread = proc_alloc_thread();
29  p->main_thread->tid = p->pid;
30  p->main_thread->process = p;
31 
32  p->main_thread->kstack = zoner_new_zone(VMM_PAGE_SIZE);
33  if (!p->main_thread->kstack.start) {
34  return -ENOMEM;
35  }
36  _thread_setup_kstack(p->main_thread, p->main_thread->kstack.start + VMM_PAGE_SIZE);
37 
38  /* setting current work directory */
39  p->cwd = 0;
40 
41  p->fds = 0;
42 
43  /* setting signal handlers to 0 */
44  p->main_thread->signals_mask = 0x0; /* All signals are disabled. */
45  p->main_thread->pending_signals_mask = 0x0;
46  memset((void*)p->main_thread->signal_handlers, 0, sizeof(p->main_thread->signal_handlers));
47 
48  return 0;
49 }
