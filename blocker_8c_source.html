100 int should_unblock_select_block(thread_t* thread)
101 {
102  if (thread->unblock_time != 0 && thread->unblock_time <= timeman_now()) {
103  return true;
104  }
105 
106  file_descriptor_t* fd;
107  for (int i = 0; i < thread->nfds; i++) {
108  if (FD_ISSET(i, &thread->readfds)) {
109  fd = proc_get_fd(thread->process, i);
110  if (fd->ops->can_read(fd->dentry, fd->offset)) {
111  return true;
112  }
113  }
114  }
115 
116  for (int i = 0; i < thread->nfds; i++) {
117  if (FD_ISSET(i, &thread->writefds)) {
118  fd = proc_get_fd(thread->process, i);
119  if (fd->ops->can_write(fd->dentry, fd->offset)) {
120  return true;
121  }
122  }
123  }
124  return false;
125 }
