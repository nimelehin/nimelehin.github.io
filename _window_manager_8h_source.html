29  inline void add_window(Window* window)

39  inline void remove_window(Window* window)
40  {
41  if (movable_window().ptr() == window) {
42  m_movable_window = nullptr;
43  }
44  if (active_window().ptr() == window) {
45  m_active_window = nullptr;
46  }
47  if (hovered_window().ptr() == window) {
48  m_hovered_window = nullptr;
49  }
50  m_windows.remove(window);
51  m_compositor.invalidate(window->bounds());
52  notify_window_status_changed(window->id(), WindowStatusUpdateType::Removed);
53  delete window;
54  }
