90  [[gnu::always_inline]] inline void mix_with(const Color& clr)
91  {
92  if (clr.is_opaque() || !alpha()) {
93  return;
94  }
95 
96  int alpha_c = 256 * (alpha() + clr.alpha()) - alpha() * clr.alpha();
97  int alpha_of_me = alpha() * (255 - clr.alpha());
98  int alpha_of_it = 256 * clr.alpha();
99 
100  m_r = (red() * alpha_of_me + clr.red() * alpha_of_it) / alpha_c;
101  m_g = (green() * alpha_of_me + clr.green() * alpha_of_it) / alpha_c;
102  m_b = (blue() * alpha_of_me + clr.blue() * alpha_of_it) / alpha_c;
103  m_opacity = 255 - (alpha_c / 256);
104  }
