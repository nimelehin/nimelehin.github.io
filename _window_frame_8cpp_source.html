117 void WindowFrame::draw(LG::Context& ctx)
118 {
119  if (!visible()) {
120  return;
121  }
122  int x = m_window.bounds().min_x();
123  int y = m_window.bounds().min_y();
124  size_t width = m_window.bounds().width();
125  size_t height = m_window.bounds().height();
126 
127  int right_x = x + width - right_border_size();
128  int bottom_y = y + height - bottom_border_size();
129 
130  // Drawing frame
131  ctx.set_fill_color(color());
132  ctx.fill(LG::Rect(x, y, width, top_border_size()));
133  ctx.fill(LG::Rect(x, y, left_border_size(), height));
134  ctx.fill(LG::Rect(right_x, y, right_border_size(), height));
135  ctx.fill(LG::Rect(x, bottom_y, width, bottom_border_size()));
136 
137  // Drawing labels, icons.
138  // Drawing positions are calculated using a start of the frame.
139  ctx.set_fill_color(LG::Color::White);
140  ctx.draw({ x + spacing(), y + icon_y_offset() }, icon());
141 
142  constexpr int start_controls_offset = icon_width() + 2 * spacing();
143  int start_controls = x + start_controls_offset;
144  for (int i = 0; i < m_control_panel_buttons.size(); i++) {
145  m_control_panel_buttons[i]->display(ctx, { start_controls, y + text_y_offset() });
146  start_controls += spacing() + m_control_panel_buttons[i]->bounds().width();
147  }
148 
149  int start_buttons = right_x - spacing() - m_window_control_buttons[0]->bounds().width();
150  for (int i = 0; i < m_window_control_buttons.size(); i++) {
151  m_window_control_buttons[i]->display(ctx, { start_buttons, y + button_y_offset() });
152  start_buttons += -spacing() - m_window_control_buttons[i]->bounds().width();
153  }
154 }

156 void WindowFrame::receive_tap_event(const LG::Point<int>& tap)
157 {
158  // Calculating buttons' positions
159  size_t width = m_window.bounds().width();
160  int right_x = width - right_border_size();
161  int start_buttons = right_x - spacing() - m_window_control_buttons[0]->bounds().width();
162  for (int button_id = 0; button_id < m_window_control_buttons.size(); button_id++) {
163  auto button_frame = m_window_control_buttons[button_id]->bounds();
164  button_frame.offset_by(start_buttons, button_y_offset());
165  if (button_frame.contains(tap)) {
166  handle_control_panel_tap(button_id);
167  return;
168  }
169  start_buttons += -spacing() - button_frame.width();
170  }
171 }
