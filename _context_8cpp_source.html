135 void Context::add_ellipse(const Rect& rect)
136 {
137  int rx = rect.width() / 2;
138  int ry = rect.height() / 2;
139  int xc = rect.mid_x();
140  int yc = rect.mid_y();
141 
142  double dx, dy, d1, d2, x, y;
143  double tmp_d1, tmp_d2;
144  x = 0;
145  y = ry;
146 
147  d1 = (ry * ry) - (rx * rx * ry) + (0.25 * rx * rx);
148  dx = 2 * ry * ry * x;
149  dy = 2 * rx * rx * y;
150 
151  while (dx < dy) {
152  m_bitmap[y + yc][(int)x + xc] = fill_color();
153  m_bitmap[y + yc][(int)-x + xc] = fill_color();
154  m_bitmap[-y + yc][(int)x + xc] = fill_color();
155  m_bitmap[-y + yc][(int)-x + xc] = fill_color();
156 
157  x++;
158  dx += 2 * ry * ry;
159  tmp_d1 = d1;
160  d1 += ry * ry + dx;
161  if (tmp_d1 >= 0) {
162  y--;
163  dy -= 2 * rx * rx;
164  d1 -= dy;
165  }
166  }
167 
168  d2 = ((ry * ry) * ((x + 0.5) * (x + 0.5))) + ((rx * rx) * ((y - 1) * (y - 1))) - (rx * rx * ry * ry);
169 
170  while (y >= 0) {
171  m_bitmap[y + yc][(int)x + xc] = fill_color();
172  m_bitmap[y + yc][(int)-x + xc] = fill_color();
173  m_bitmap[-y + yc][(int)x + xc] = fill_color();
174  m_bitmap[-y + yc][(int)-x + xc] = fill_color();
175 
176  y--;
177  dy -= 2 * rx * rx;
178  tmp_d2 = d2;
179  d2 += rx * rx - dy;
180  if (tmp_d2 <= 0) {
181  x++;
182  dx += 2 * ry * ry;
183  d2 += dx;
184  }
185  }
186 }
