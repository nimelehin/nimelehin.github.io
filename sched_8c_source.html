173 void sched()
174 {
175  for (;;) {
176  while (!_master_buf[_buf_read_prio].head) {
177  if (_buf_read_prio >= MIN_PRIO) {
178  tasking_kill_dying();
179  sched_unblock_threads();
180  _sched_swap_buffers();
181  } else {
182  _buf_read_prio++;
183  }
184  }
185 
186  thread_t* thread = _master_buf[_buf_read_prio].head;
187  _master_buf[_buf_read_prio].head = thread->sched_next;
188  if (_master_buf[_buf_read_prio].tail == thread) {
189  _master_buf[_buf_read_prio].tail = 0;
190  }
191  thread->sched_next = thread->sched_prev = 0;
192 #ifdef SCHED_DEBUG
193  log("next to run %d %x\n", thread->tid, thread->tf->eip);
194 #endif
195 #ifdef SCHED_SHOW_STAT
196  log("[STAT] procs in buffer: %d", _debug_count_of_proc_in_buf(_master_buf));
197 #endif
198  ASSERT(thread->status == THREAD_RUNNING);
199  switchuvm(thread);
200  switch_contexts(&THIS_CPU->scheduler, thread->context);
201  }
202 }
