112 void View::receive_mouse_move_event(MouseEvent& event)
113 {
114  if (!is_hovered()) {
115  m_hovered = true;
116  hover_begin(LG::Point<int>(event.x(), event.y()));
117  }
118 
119  foreach_subview([&](View& subview) -> bool {
120  if (subview.is_hovered() && !subview.frame().contains(event.x(), event.y())) {
121  LG::Point<int> point(event.x(), event.y());
122  point.offset_by(-subview.frame().origin());
123  MouseLeaveEvent mle(point.x(), point.y());
124  subview.receive_mouse_leave_event(mle);
125  } else if (!subview.is_hovered() && subview.frame().contains(event.x(), event.y())) {
126  LG::Point<int> point(event.x(), event.y());
127  point.offset_by(-subview.frame().origin());

130  }
131  return true;
132  });
