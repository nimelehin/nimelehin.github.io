45 UniquePtr<Message> ClientDecoder::handle(const KeyboardMessage& msg)
46 {
47  if (App::the().window().id() == msg.win_id()) {
48  // Checking if the key is down or up
49  if (msg.kbd_key() >> 31) {
50  uint32_t key = msg.kbd_key();
51  key &= 0xEFFFFFFF;
52  m_event_loop.add(App::the().window(), new KeyUpEvent(key));
53  } else {
54  m_event_loop.add(App::the().window(), new KeyDownEvent(msg.kbd_key()));
55  }
56  }
57  return nullptr;
58 }
