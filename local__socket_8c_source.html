69 {

71 
72  uint32_t file_mode = S_IFSOCK | S_IRUSR | S_IXUSR | S_IRGRP | S_IXGRP | S_IROTH | S_IXOTH;
73  vfs_create(p->cwd, name, len, file_mode);

96 int local_socket_connect(file_descriptor_t* sock, char* name, uint32_t len)
97 {
98  proc_t* p = RUNNIG_THREAD->process;
99 
100  dentry_t* bind_dentry;
101  int res = vfs_resolve_path_start_from(p->cwd, name, &bind_dentry);
102  if (res < 0) {
103 #ifdef LOCAL_SOCKET_DEBUG
104  log_error("Connect: can't find path to file : %d pid\n", p->pid);
105 #endif
106  return res;
107  }
108  if ((bind_dentry->inode->mode & S_IFSOCK) == 0) {
109 #ifdef LOCAL_SOCKET_DEBUG
110  log_error("Connect: file not a socket : %d pid\n", p->pid);
111 #endif
112  return -ENOTSOCK;
113  }
114 
115  if (!bind_dentry->sock) {
116  return -EBADF;
117  }
118  sock->sock_entry = socket_duplicate(bind_dentry->sock);
119  sock->offset = bind_dentry->sock->buffer.end; /* Starting to read from the end */
120 #ifdef LOCAL_SOCKET_DEBUG
121  log("Connected to local socket at %x : %d pid", bind_dentry->sock, p->pid);
122 #endif
123  return 0;
