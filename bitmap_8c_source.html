33 int bitmap_find_space(bitmap_t bitmap, int req)
34 {
35  int taken = 0;
36  int start = 0;
37  for (int i = 0; i < bitmap.len; i++) {
38  if (bitmap.data[i] == 0xff) {
39  continue;
40  }
41  for (int j = 0; j < 8; j++) {
42  if ((bitmap.data[i] >> j) & 1) {
43  taken = 0;
44  } else {
45  if (taken == 0) {
46  start = i * 8 + j;
47  }
48  taken++;
49  if (taken == req) {
50  return start;
51  }
52  }
53  }
54  }
55  return -ENODATA;
56 }
