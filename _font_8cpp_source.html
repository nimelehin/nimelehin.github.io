79 Font* Font::load_from_mem(uint8_t* font_data)
80 {
81  if (!font_data) {
82  return nullptr;
83  }
84 
85  FontFileHeader& header = *(FontFileHeader*)font_data;
86 
87  if (memcmp((uint8_t*)header.magic, (const uint8_t*)"!Fnt", 4)) {
88  write(1, "Font unsupported\n", 17);
89  return nullptr;
90  }
91 
92  size_t count = 0;
93  if (header.type == 0) {
94  count = 256;
95  } else if (header.type == 1) {
96  count = 384;
97  } else {
98  write(1, "Type unsupported\n", 17);
99  return nullptr;
100  }
101 
102  size_t bytes_per_glyph = sizeof(uint32_t) * header.glyph_height;
103 
104  uint32_t* raw_data = (uint32_t*)(font_data + sizeof(FontFileHeader));
105  uint8_t* width_data = nullptr;
106  if (header.is_variable_width) {
107  width_data = (uint8_t*)((uint8_t*)(raw_data) + count * bytes_per_glyph);
108  }
109 
110  return new Font(raw_data, width_data, header.glyph_width, header.glyph_height, count, header.is_variable_width);
111 }
