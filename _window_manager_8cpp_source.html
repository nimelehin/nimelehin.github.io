46 bool WindowManager::continue_window_move(MouseEvent* mouse_event)
47 {
48  if (!movable_window()) {
49  return false;
50  }
51 
52  update_mouse_position(mouse_event);
53 
54  if (!is_mouse_left_button_pressed()) {
55  m_movable_window = nullptr;
56  return true;
57  }
58 
59  auto bounds = movable_window()->bounds();
60  m_compositor.invalidate(movable_window()->bounds());
61  movable_window()->bounds().offset_by(mouse_event->packet().x_offset, -mouse_event->packet().y_offset);
62  movable_window()->content_bounds().offset_by(mouse_event->packet().x_offset, -mouse_event->packet().y_offset);
63  bounds.unite(movable_window()->bounds());
64  m_compositor.invalidate(bounds);
65  return true;
66 }

68 void WindowManager::update_mouse_position(MouseEvent* mouse_event)
69 {
70  auto invalidate_bounds = m_compositor.cursor_manager().current_cursor().bounds();
71  invalidate_bounds.origin().set(m_compositor.cursor_manager().draw_position(m_mouse_x, m_mouse_y));
72  m_compositor.invalidate(invalidate_bounds);
73 
74  m_mouse_x += mouse_event->packet().x_offset;
75  m_mouse_y -= mouse_event->packet().y_offset;
76  if (m_mouse_x < 0) {
77  m_mouse_x = 0;
78  }
79  if (m_mouse_y < 0) {
80  m_mouse_y = 0;
81  }
82  if (m_mouse_x >= m_screen.width() - 1) {
83  m_mouse_x = m_screen.width() - 1;
84  }
85  if (m_mouse_y >= m_screen.height() - 1) {
86  m_mouse_y = m_screen.height() - 1;
87  }
88 
89  invalidate_bounds.origin().set(m_compositor.cursor_manager().draw_position(m_mouse_x, m_mouse_y));
90  m_compositor.invalidate(invalidate_bounds);
91 
92  m_mouse_changed_button_status = false;
93  if (m_mouse_left_button_pressed != (mouse_event->packet().button_states & 1)) {
94  m_mouse_changed_button_status = true;
95  }
96 
97  m_mouse_left_button_pressed = (mouse_event->packet().button_states & 1);
98 }

100 void WindowManager::receive_mouse_event(UniquePtr<LFoundation::Event> event)
101 {
102  MouseEvent* mouse_event = (MouseEvent*)event.release();
103  auto new_hovered_window = WeakPtr<Window>();
104 
105  if (continue_window_move(mouse_event)) {
106  goto end;
107  }
108 
109  update_mouse_position(mouse_event);
110 
111  for (auto& window : m_windows) {
112  if (window.bounds().contains(m_mouse_x, m_mouse_y)) {
113  if (window.frame().bounds().contains(m_mouse_x, m_mouse_y)) {
114  if (is_mouse_left_button_pressed()) {
115  auto tap_point = LG::Point<int>(m_mouse_x - window.frame().bounds().min_x(), m_mouse_y - window.frame().bounds().min_y());
116  window.frame().receive_tap_event(tap_point);
117  start_window_move(window);
118  }
119  } else if (window.content_bounds().contains(m_mouse_x, m_mouse_y)) {
120  LG::Point<int> point(m_mouse_x, m_mouse_y);
121  point.offset_by(-window.content_bounds().origin());
122  m_event_loop.add(m_connection, new SendEvent(new MouseMoveMessage(window.connection_id(), window.id(), point.x(), point.y())));
123  new_hovered_window = window.weak_ptr();
124 
125  if (m_mouse_changed_button_status) {
126  // FIXME: only left button for now!
127  m_event_loop.add(m_connection, new SendEvent(new MouseActionMessage(window.connection_id(), window.id(), !is_mouse_left_button_pressed(), point.x(), point.y())));
128  }
129  }
130 
131  if (is_mouse_left_button_pressed() && m_active_window.ptr() != &window) {
132  bring_to_front(window);
133  m_active_window = window.weak_ptr();
134  }
135 
136  break;
137  }
138  }
139 
140  if (hovered_window()) {
141  auto window = hovered_window();
142  if (window->id() != new_hovered_window->id()) {
143  m_event_loop.add(m_connection, new SendEvent(new MouseLeaveMessage(window->connection_id(), window->id(), 0, 0)));
144  }
145  }
146 
147  m_hovered_window = new_hovered_window;
148 
149 end:
150  delete mouse_event;
151 }
