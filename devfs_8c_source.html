61 static devfs_inode_t* _devfs_get_entry(uint32_t indx)
62 {
63  uint32_t entries_per_zone = DEVFS_ZONE_SIZE / DEVFS_INODE_LEN;
64  uint32_t zones_count = entry_zones.size;
65  if (indx >= zones_count * entries_per_zone) {
66  return 0;
67  }
68 
69  uint32_t requested_zone = indx / entries_per_zone;
70  uint32_t index_within_zone = indx % entries_per_zone;
71  uint32_t* tmp_ptr = dynamic_array_get(&entry_zones, requested_zone);
72  devfs_inode_t* res = (devfs_inode_t*)*tmp_ptr;
73  if (!res) {
74  return 0;
75  }
76  return &res[index_within_zone];
77 }

83 static devfs_inode_t* _devfs_new_entry()
84 {
85  if (free_space_in_last_entry_zone < sizeof(devfs_inode_t)) {
86  _devfs_alloc_entry_zone();
87  }
88 
89  devfs_inode_t* res = (devfs_inode_t*)next_space_to_put_entry;
90  res->index = next_inode_index++;
91  next_space_to_put_entry += sizeof(devfs_inode_t);
92  free_space_in_last_entry_zone -= sizeof(devfs_inode_t);
93  return res;
94 }

121 static int _devfs_add_to_list(devfs_inode_t* parent, devfs_inode_t* new_entry)
122 {
123  if (!parent || !new_entry) {
124  return -EINVAL;
125  }
126 
127  if (!parent->last) {
128  parent->first = new_entry;
129  parent->last = new_entry;
130  new_entry->prev = 0;
131  new_entry->next = 0;
132  } else {
133  parent->last->next = new_entry;
134  new_entry->prev = parent->last;
135  parent->last = new_entry;
136  new_entry->next = 0;
137  }
138 
139  new_entry->parent = parent;
140 
141  return 0;
142 }

144 static devfs_inode_t* _devfs_alloc_entry(devfs_inode_t* parent)
145 {
146  devfs_inode_t* new_entry = _devfs_new_entry();
147  if (!new_entry) {

153  free_space_in_last_entry_zone += sizeof(devfs_inode_t);
154  return 0;

240 int devfs_getdents(dentry_t* dir, uint8_t* buf, uint32_t* offset, uint32_t len)
241 {
242  devfs_inode_t* devfs_inode = (devfs_inode_t*)dir->inode;
243 
244  /* Currently we leave, dir when offset is the max number. */
245  if (*offset == 0xffffffff) {
246  return 0;
247  }
248 
249  int already_read = 0;
250  dirent_t tmp;
251  tmp.name = 0;
252 
253  /* Return . */
254  if (*offset == 0) {
255  int rec_len = 8 + 1 + 1;
256 
257  if (len < rec_len) {
258  if (already_read) {
259  return already_read;
260  } else {
261  return -EINVAL;
262  }
263  }
264 
265  tmp.inode = devfs_inode->index;
266  tmp.rec_len = rec_len;
267  tmp.name_len = 1;
268  memcpy(buf + already_read, (void*)&tmp, 8);
269  memcpy(buf + already_read + 8, ".", 1);
270  buf[already_read + rec_len - 1] = '\0';
271  already_read += rec_len;
272  len -= rec_len;
273  *offset = 1;
274  }
275 
276  /* Return .. */
277  if (*offset == 1) {
278  int rec_len = 8 + 2 + 1;
279 
280  if (len < rec_len) {
281  if (already_read) {
282  return already_read;
283  } else {
284  return -EINVAL;
285  }
286  }
287 
288  if (devfs_inode->parent) {
289  tmp.inode = devfs_inode->parent->index;
290  } else {
291  tmp.inode = devfs_inode->index;
292  }
293  tmp.rec_len = rec_len;
294  tmp.name_len = 2;
295  memcpy(buf + already_read, (void*)&tmp, 8);
296  memcpy(buf + already_read + 8, "..", 2);
297  buf[already_read + rec_len - 1] = '\0';
298  already_read += rec_len;
299  len -= rec_len;
300  *offset = 2;
301  }
302 
303  /* Scanining dir from the start */
304  if (*offset == 2) {
305  if (!devfs_inode->first) {
306  return 0;
307  }
308  *offset = (uint32_t)devfs_inode->first;
309  }
310 
311  while (*offset != 0xffffffff) {
312  devfs_inode_t* child_devfs_inode = (devfs_inode_t*)*offset;
313  uint32_t name_len = strlen(child_devfs_inode->name);
314  int rec_len = 8 + name_len + 1;
315 
316  if (len < rec_len) {
317  if (already_read) {
318  return already_read;
319  } else {
320  return -EINVAL;
321  }
322  }
323 
324  tmp.inode = child_devfs_inode->index;
325  tmp.rec_len = rec_len;
326  tmp.name_len = name_len;
327  memcpy(buf + already_read, (void*)&tmp, 8);
328  memcpy(buf + already_read + 8, child_devfs_inode->name, name_len);
329  buf[already_read + rec_len - 1] = '\0';
330  already_read += rec_len;
331  len -= rec_len;
332  if (child_devfs_inode->next) {
333  *offset = (uint32_t)child_devfs_inode->next;
334  } else {
335  *offset = 0xffffffff;
336  }
337  }
338 
339  return already_read;
340 }

342 int devfs_lookup(dentry_t* dir, const char* name, uint32_t len, uint32_t* res_inode_indx)
343 {
344  devfs_inode_t* devfs_inode = (devfs_inode_t*)dir->inode;
345 
346  if (len == 1) {
347  if (name[0] == '.') {
348  *res_inode_indx = devfs_inode->index;
349  return 0;
350  }
351  }
352 
353  if (len == 2) {
354  if (name[0] == '.' && name[1] == '.') {
355  *res_inode_indx = devfs_inode->parent->index;
356  return 0;
357  }
358  }
359 
360  devfs_inode_t* child = devfs_inode->first;
361  while (child) {
362  uint32_t child_name_len = strlen(child->name);
363  if (len == child_name_len) {
364  if (strncmp(name, child->name, len) == 0) {
365  *res_inode_indx = child->index;
366  return 0;
367  }
368  }
369  child = child->next;
370  }
371 
372  return -ENOENT;
373 }
