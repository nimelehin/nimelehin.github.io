53 void* kmalloc(uint32_t size)
54 {
55  int act_size = size + sizeof(kmalloc_header_t);
56 
57  int blocks_needed = (act_size + KMALLOC_BLOCK_SIZE - 1) / KMALLOC_BLOCK_SIZE;
58 
59  int start = bitmap_find_space(bitmap, blocks_needed);
60  if (start < 0) {
61  kprintf("[Err] NO SPACE AT KMALLOC");
62  while (1) {}
63  return 0;
64  }
65 
66  kmalloc_header_t* space = (kmalloc_header_t*)kmalloc_to_vaddr(start);
67  space->len = act_size;
68  bitmap_set_range(bitmap, start, blocks_needed);
69 
70  return (void*)&space[1];
71 }
