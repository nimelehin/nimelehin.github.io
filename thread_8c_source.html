22 int _thread_setup_kstack(thread_t* thread, uint32_t esp)
23 {
24  char* sp = (char*)(esp);
25 
26  /* setting trapframe in kernel stack */
27  sp -= sizeof(*thread->tf);
28  thread->tf = (trapframe_t*)sp;
29 
30  /* setting return point in kernel stack */
31  sp -= 4;
32  *(uint32_t*)sp = (uint32_t)trap_return;
33 
34  /* setting context in kernel stack */
35  sp -= sizeof(*thread->context);
36  thread->context = (context_t*)sp;
37 
38  /* setting init data */
39  memset((void*)thread->context, 0, sizeof(*thread->context));
40  thread->context->eip = (uint32_t)_tasking_jumper;
41  memset((void*)thread->tf, 0, sizeof(*thread->tf));
42 
43  /* setting fpu */
44  thread->fpu_state = kmalloc_aligned(sizeof(fpu_state_t), 16);
45  fpu_reset_state(thread->fpu_state);
46 
47  return 0;
48 }

139 int thread_fill_up_stack(thread_t* thread, int argc, char** argv, char** env)
140 {
141  /* TODO: Add env */
142  uint32_t argv_data_size = 0;
143  for (int i = 0; i < argc; i++) {
144  argv_data_size += strlen(argv[i]) + 1;
145  }
146 
147  if (argv_data_size % 4) {
148  argv_data_size += 4 - (argv_data_size % 4);
149  }
150 
151  uint32_t size_in_stack = argv_data_size + (argc + 1) * sizeof(char*) + sizeof(argc) + sizeof(char*);
152  int* tmp_buf = (int*)kmalloc(size_in_stack);
153  if (!tmp_buf) {
154  return -EAGAIN;
155  }
156  memset((void*)tmp_buf, 0, size_in_stack);
157 
158  char* tmp_buf_ptr = ((char*)tmp_buf) + size_in_stack;
159  char* tmp_buf_data_ptr = tmp_buf_ptr - argv_data_size;
160  uint32_t* tmp_buf_array_ptr = (uint32_t*)((char*)tmp_buf_data_ptr - (argc + 1) * sizeof(char*));
161  int* tmp_buf_argv_ptr = (int*)((char*)tmp_buf_array_ptr - sizeof(char*));
162  int* tmp_buf_argc_ptr = (int*)((char*)tmp_buf_argv_ptr - sizeof(int));
163 
164  uint32_t data_esp = get_stack_pointer(thread->tf) - argv_data_size;
165  uint32_t array_esp = data_esp - (argc + 1) * sizeof(char*);
166  uint32_t argv_esp = array_esp - 4;
167  uint32_t argc_esp = argv_esp - 4;
168 
169  for (int i = argc - 1; i >= 0; i--) {
170  uint32_t len = strlen(argv[i]);
171  tmp_buf_ptr -= len + 1;
172  thread->tf->esp -= len + 1;
173  memcpy(tmp_buf_ptr, argv[i], len);
174  tmp_buf_ptr[len] = 0;
175 
176  tmp_buf_array_ptr[i] = get_stack_pointer(thread->tf);
177  }
178  tmp_buf_array_ptr[argc] = 0;
179 
180  *tmp_buf_argv_ptr = array_esp;
181  *tmp_buf_argc_ptr = argc;
182 
183  set_stack_pointer(thread->tf, argc_esp);
184 
185  vmm_copy_to_pdir(thread->process->pdir, (uint8_t*)tmp_buf, get_stack_pointer(thread->tf), size_in_stack);
186 
187  kfree(tmp_buf);
188 
189  return 0;
190 }
