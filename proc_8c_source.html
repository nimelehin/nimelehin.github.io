66 {

69  p->is_kthread = false;
70 
71  p->main_thread = proc_alloc_thread();
72  int res = thread_setup_main(p, p->main_thread);
73  if (res != 0) {
74  return res;
75  }

78  p->cwd = 0;
79 
80  /* allocating space for open files */

156 static int _proc_load_bin(proc_t* p, file_descriptor_t* fd)
157 {
158  uint32_t code_size = fd->dentry->inode->size;
159  log("Load with size %x", code_size);
160  proc_zone_t* code_zone = proc_new_random_zone(p, code_size);
161  code_zone->type = ZONE_TYPE_CODE;
162  code_zone->flags |= ZONE_READABLE | ZONE_EXECUTABLE;
163 
164  /* THIS IS FOR BSS WHICH COULD BE IN THIS ZONE */
165  code_zone->flags |= ZONE_WRITABLE;
166 
167  proc_zone_t* bss_zone = proc_new_random_zone(p, 2 * 4096);
168  bss_zone->type = ZONE_TYPE_DATA;
169  bss_zone->flags |= ZONE_READABLE | ZONE_WRITABLE;
170 
171  proc_zone_t* stack_zone = proc_new_random_zone_backward(p, VMM_PAGE_SIZE);
172  stack_zone->type = ZONE_TYPE_STACK;
173  stack_zone->flags |= ZONE_READABLE | ZONE_WRITABLE;
174 
175  /* Copying an exec code */
176  uint8_t* prog = kmalloc(fd->dentry->inode->size);
177  fd->ops->read(fd->dentry, prog, 0, fd->dentry->inode->size);
178  vmm_copy_to_pdir(p->pdir, prog, code_zone->start, fd->dentry->inode->size);
179 
180  /* Setting registers */
181  thread_t* main_thread = p->main_thread;
182  thread_set_stack(main_thread, stack_zone->start + VMM_PAGE_SIZE, stack_zone->start + VMM_PAGE_SIZE);
183  thread_set_eip(main_thread, code_zone->start);
184 
185  if (code_size == 0x135C) {
186  thread_set_eip(main_thread, 0x80);
187  }
188  kfree(prog);
189 
190  return 0;
191 }

193 int proc_load(proc_t* p, const char* path)
194 {
195  dentry_t* file;
196  file_descriptor_t fd;
197 
198  if (vfs_resolve_path_start_from(p->cwd, path, &file) < 0) {
199  return -ENOENT;
200  }
201  if (vfs_open(file, &fd, O_RDONLY) < 0) {
202  dentry_put(file);
203  return -ENOENT;
204  }
205 
206  pdirectory_t* old_pdir = p->pdir;
207  pdirectory_t* new_pdir = vmm_new_user_pdir();
208  vmm_switch_pdir(new_pdir);
209  p->pdir = new_pdir;
210  int err = _proc_load_bin(p, &fd);
211 
212  if (!err) {
213  if (!p->cwd) {
214  p->cwd = dentry_get_parent(file);
215  }
216  vmm_free_pdir(old_pdir);
217  } else {
218  p->pdir = old_pdir;
219  vmm_switch_pdir(old_pdir);
220  vmm_free_pdir(new_pdir);
221  }
222 
223  dentry_put(file);
224  vfs_close(&fd);
225  return err;
226 }

232 int proc_free(proc_t* p)
233 {
234  if (p->status == PROC_DEAD || p->status == PROC_INVALID || p->status == PROC_ALIVE || p->pid == 0) {
235  return -ESRCH;
236  }
237 
238  /* closing opend fds */
239  if (p->fds) {
240  for (int i = 0; i < MAX_OPENED_FILES; i++) {
241  if (p->fds[i].dentry) {
242  /* think as an active fd */
243  vfs_close(&p->fds[i]);
244  }
245  }
246  kfree(p->fds);
247  }
248 
249  if (p->cwd) {
250  dentry_put(p->cwd);
251  }
252 
253  dynamic_array_free(&p->zones);
254 
255  /* Key parts deletion. After that line you can't work with this process. */
256  proc_kill_all_threads(p);
257  p->pid = 0;
258 
259  if (!p->is_kthread) {
260  vmm_free_pdir(p->pdir);
261  p->pdir = 0;
262  }
263 
264  return 0;
265 }

313 int proc_chdir(proc_t* p, const char* path)
314 {
315  char* kpath = 0;
316  if (!str_validate_len(path, 128)) {
317  return -EINVAL;
318  }
319  kpath = kmem_bring_to_kernel(path, strlen(path) + 1);
320 
321  dentry_t* new_cwd;
322  int ret = vfs_resolve_path_start_from(p->cwd, kpath, &new_cwd);
323  if (ret) {
324  return -ENOENT;
325  }
326 
327  if (!dentry_inode_test_flag(new_cwd, S_IFDIR)) {
328  dentry_put(new_cwd);
329  return -ENOTDIR;
330  }
331 
332  /* Put an old one back */
333  if (p->cwd) {
334  dentry_put(p->cwd);
335  }
336  p->cwd = new_cwd;
337  return 0;
