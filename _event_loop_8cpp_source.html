37  if (m_waiting_fds[i].m_on_read) {
38  FD_SET(m_waiting_fds[i].m_fd, &readfds);
39  }
40  if (m_waiting_fds[i].m_on_write) {
41  FD_SET(m_waiting_fds[i].m_fd, &writefds);
42  }
43  if (nfds < m_waiting_fds[i].m_fd) {
44  nfds = m_waiting_fds[i].m_fd;
45  }
46  }
47 
48  // For now, that means, that we don't wait for fds.
49  timeval_t timeout;
50  timeout.tv_sec = 0;
51  timeout.tv_usec = 0;
52 
53  int res = select(nfds + 1, &readfds, &writefds, nullptr, &timeout);
54 
55  for (int i = 0; i < m_waiting_fds.size(); i++) {
56  if (m_waiting_fds[i].m_on_read) {
57  if (FD_ISSET(m_waiting_fds[i].m_fd, &readfds)) {
58  m_event_queue.push_back(QueuedEvent(m_waiting_fds[i], new FDWaiterReadEvent()));
59  }
60  }

78 void EventLoop::pump()
79 {
80  check_fds();
81  check_timers();
82  Vector<QueuedEvent> events_to_dispatch(move(m_event_queue));
83  for (int i = 0; i < events_to_dispatch.size(); i++) {
84  events_to_dispatch[i].receiver.receive_event(move(events_to_dispatch[i].event));
85  }
