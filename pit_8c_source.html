20 static int _pit_set_frequency(uint16_t freq)
21 {
22  system_disable_interrupts();
23  uint32_t divisor = PIT_BASE_FREQ / freq;
24  if (divisor > 0xffff) {
25  return -1;
26  }
27  uint8_t low = (uint8_t)(divisor & 0xFF);
28  uint8_t high = (uint8_t)((divisor >> 8) & 0xFF);
29  port_byte_out(0x43, 0x36); // 0b110110
30  port_byte_out(0x40, low);
31  port_byte_out(0x40, high);
32  system_enable_interrupts();
33  return 0;
34 }
