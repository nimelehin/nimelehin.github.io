362 /* Keyboard interrupt handler */
363 void keyboard_handler()
364 {
365  uint32_t scancode = (uint32_t)port_byte_in(0x60);
366  if (scancode == 0xE0) {
367  _kbdriver_has_prefix_e0 = true;
368  return;
369  }
370  if (_kbdriver_has_prefix_e0) {
371  scancode |= 0x100;
372  _kbdriver_has_prefix_e0 = false;
373  }
374 
375  key_t key;
376  kbd_packet_t packet;
377 
378  if (scancode & 0x80) {
379  scancode -= 0x80;
380  key = _kbdriver_get_keycode(scancode);
381  packet.key = key | (1 << 31);
382  switch (key) {
383  case KEY_LCTRL:
384  case KEY_RCTRL:
385  _kbdriver_ctrl_enabled = false;
386  break;
387  case KEY_LSHIFT:
388  case KEY_RSHIFT:
389  _kbdriver_shift_enabled = false;
390  break;
391  case KEY_CAPSLOCK:
392  _kbdriver_caps_enabled ^= 1;
393  break;
394  case KEY_LALT:
395  case KEY_RALT:
396  _kbdriver_alt_enabled = false;
397  break;
398  default:
399  key = _kbdriver_apply_modifiers(key);
400  packet.key = key | (1 << 31);
401  }
402  } else {
403  _kbdriver_last_scancode = scancode;
404  key = _kbdriver_get_keycode(scancode);
405  packet.key = key;
406  switch (key) {
407  case KEY_LCTRL:
408  case KEY_RCTRL:
409  _kbdriver_ctrl_enabled = true;
410  break;
411  case KEY_LSHIFT:
412  case KEY_RSHIFT:
413  _kbdriver_shift_enabled = true;
414  break;
415  case KEY_CAPSLOCK:
416  break;
417  case KEY_LALT:
418  case KEY_RALT:
419  _kbdriver_alt_enabled = true;
420  break;
421  default:
422  key = _kbdriver_apply_modifiers(key);
423  packet.key = key;
424  /* FIXME: ifdef here to support console mode */
425  /* tty_eat_key(key); */
426  }
427  }
428 
429  ringbuffer_write(&kbd_buffer, (uint8_t*)&packet, sizeof(kbd_packet_t));
430 }
