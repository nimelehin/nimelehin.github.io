122 extern int _thread_setup_kstack(thread_t* thread, uint32_t esp);
123 static int signal_setup_stack_to_handle_signal(thread_t* thread, int signo)
124 {
125  system_disable_interrupts();
126  pdirectory_t* prev_pdir = vmm_get_active_pdir();
127  vmm_switch_pdir(thread->process->pdir);
128 
129  uint32_t old_esp = thread->tf->esp;
130  uint32_t magic = MAGIC_STATE_JUST_TF; /* helps to restore thread after sgnal to the right state */
131 
132  /* TODO: Add support for SMP */
133  if (thread != RUNNIG_THREAD) {
134  /*
135  If we are here that means that the thread was stopped while
136  being in kernel (because of scheduler or blocker). That means,
137  we need not corrupt it's kernel state, so we create a new state
138  to send signal.
139 
140  We setup a new kernel state upper the previous one
141  Stack:
142  Trapframe
143  Function calls and local vars...
144  Context
145  ---- add after setup ----
146  Trapframe for signal
147  Context for signal
148  */
149 
150  magic = MAGIC_STATE_NEW_STACK;
151  trapframe_t* old_tf = thread->tf;
152  context_t* old_ctx = thread->context;
153 
154  _thread_setup_kstack(thread, (uint32_t)thread->context);
155  memcpy((void*)thread->tf, (void*)old_tf, sizeof(trapframe_t));
156 
157  signal_push_to_user_stack(thread, (uint32_t)old_tf);
158  signal_push_to_user_stack(thread, (uint32_t)old_ctx);
159  signal_push_to_user_stack(thread, (uint32_t)old_tf ^ (uint32_t)old_ctx); // checksum
160  }
161  signal_push_to_user_stack(thread, thread->tf->eflags);
162  signal_push_to_user_stack(thread, thread->tf->eip);
163  signal_push_to_user_stack(thread, thread->tf->eax);
164  signal_push_to_user_stack(thread, thread->tf->ebx);
165  signal_push_to_user_stack(thread, thread->tf->ecx);
166  signal_push_to_user_stack(thread, thread->tf->edx);
167  signal_push_to_user_stack(thread, old_esp);
168  signal_push_to_user_stack(thread, thread->tf->ebp);
169  signal_push_to_user_stack(thread, thread->tf->esi);
170  signal_push_to_user_stack(thread, thread->tf->edi);
171  signal_push_to_user_stack(thread, magic);
172  signal_push_to_user_stack(thread, (uint32_t)thread->signal_handlers[signo]);
173  signal_push_to_user_stack(thread, (uint32_t)signo);
174  signal_push_to_user_stack(thread, 0); /* fake return address */
175 
176  vmm_switch_pdir(prev_pdir);
177  system_enable_interrupts();
178  return 0;
179 }

181 int signal_restore_thread_after_handling_signal(thread_t* thread)
182 {
183  int ret = thread->tf->ebx;
184  thread->tf->esp += 12; /* cleaning 3 last pushes */
185 
186  uint32_t magic = signal_pop_from_user_stack(thread);
187  thread->tf->edi = signal_pop_from_user_stack(thread);
188  thread->tf->esi = signal_pop_from_user_stack(thread);
189  thread->tf->ebp = signal_pop_from_user_stack(thread);
190  uint32_t old_esp = signal_pop_from_user_stack(thread);
191  thread->tf->edx = signal_pop_from_user_stack(thread);
192  thread->tf->ecx = signal_pop_from_user_stack(thread);
193  thread->tf->ebx = signal_pop_from_user_stack(thread);
194  thread->tf->eax = signal_pop_from_user_stack(thread);
195  thread->tf->eip = signal_pop_from_user_stack(thread);
196  thread->tf->eflags = signal_pop_from_user_stack(thread);
197  if (magic == MAGIC_STATE_NEW_STACK) {
198  uint32_t checksum = signal_pop_from_user_stack(thread);
199  context_t* old_ctx = (context_t*)signal_pop_from_user_stack(thread);
200  trapframe_t* old_tf = (trapframe_t*)signal_pop_from_user_stack(thread);
201 
202  uint32_t calced_checksum = ((uint32_t)old_tf ^ (uint32_t)old_ctx);
203 
204  if (checksum != calced_checksum) {
205  kprintf("Killed %d: wrong signal checksum\n", thread->process->pid);
206  proc_die(thread->process);
207  resched_dont_save_context();
208  }
209 
210  thread->tf = old_tf;
211  thread->context = old_ctx;
212  }
213 
214  if (old_esp != thread->tf->esp) {
215  kpanic("ESPs are diff after signal");
216  }
217 
218  /* If our thread was blocked, that means that it already has a context in stack, we need not ot overwrite it */
219  if (thread->blocker.reason != BLOCKER_INVALID) {
220  thread->status = THREAD_BLOCKED;
221  sched_dequeue(thread);
222  resched_dont_save_context();
223  }
224 
225  /* Since we already have a context in stack, we need not ot overwrite it */
226  if (magic == MAGIC_STATE_NEW_STACK) {
227  resched_dont_save_context();
228  }
229 
230  return ret;
231 }

240 /* FIXME: Don't allow to run a signal while other is in process */
241 static int signal_process(thread_t* thread, int signo)
242 {
243  if (thread->signal_handlers[signo]) {
244  signal_setup_stack_to_handle_signal(thread, signo);
245  thread->tf->eip = _signal_jumper_zone.start;
246  return UNBLOCK;
247  } else {
248  int result = signal_default_action(signo);
249  if (result == SIGNAL_ACTION_TERMINATE) {
250  proc_die(thread->process);
251  return SKIP;
252  }
253  }
254  return -EFAULT;
255 }

257 int signal_dispatch_pending(thread_t* thread)
258 {
259  uint32_t candidate_signals = thread->pending_signals_mask & thread->signals_mask;
260 
261  if (!candidate_signals) {
262  return -EACCES;
263  }
264 
265  uint32_t signo = 1;
266  for (; signo < 32; signo++) {
267  if (candidate_signals & (1 << signo)) {
268  break;
269  }
270  }
271 
272  signal_rem_pending(thread, signo);
273  int ret = signal_process(thread, signo);
274 
275  if (ret < 0) {
276  return ret;
277  }
278 
279  if (ret == UNBLOCK) {
280  if (thread->status == THREAD_BLOCKED) {
281  if (!thread->blocker.should_unblock_for_signal) {
282  return -1;
283  }
284  thread->status = THREAD_RUNNING;
285  sched_enqueue(thread);
286  }
287  }
288 
289  return 0;
290 }
