31 static uint8_t _ata_gen_drive_head_register(bool is_lba, bool is_master, uint8_t head)
32 {
33  uint8_t res = 0xA0;
34  res |= ((is_lba & 0x1) << 6);
35  res |= ((is_master & 0x1) << 4);
36  res |= (head & 15);
37  return res;
38 }

91 bool ata_indentify(ata_t* ata)
92 {
93  port_8bit_out(ata->port.device, ata->is_master ? 0xA0 : 0xB0);
94  port_8bit_out(ata->port.sector_count, 0);
95  port_8bit_out(ata->port.lba_lo, 0);
96  port_8bit_out(ata->port.lba_mid, 0);
97  port_8bit_out(ata->port.lba_hi, 0);
98  port_8bit_out(ata->port.command, 0xEC);
99 
100  // check the acceptance of a command
101  uint8_t status = port_8bit_in(ata->port.command);
102  if (status == 0x00) {
103  kprintf("Cmd isn't accepted");
104  return false;
105  }
106 
107  // waiting for processing
108  // while BSY is on
109  while ((status & 0x80) == 0x80) {
110  status = port_8bit_in(ata->port.command);
111  }
112  // check if drive isn't ready to transer DRQ
113  if ((status & 0x08) != 0x08) {
114  kprintf("Doesn't ready to transfer DRQ");
115  return false;
116  }
117 
118  // transfering 256 bytes of data
119  for (int i = 0; i < 256; i++) {
120  uint16_t data = port_16bit_in(ata->port.data);
121  if (i == 1) {
122  ata->cylindres = data;
123  }
124  if (i == 3) {
125  ata->heads = data;
126  }
127  if (i == 6) {
128  ata->sectors = data;
129  }
130  if (i == 49) {
131  if ((data >> 8) & 0x1 == 1) {
132  ata->dma = true;
133  }
134  if ((data >> 9) & 0x1 == 1) {
135  ata->lba = true;
136  }
137  }
138 
139  if (i == 60) {
140  ata->capacity = data;
141  }
142  if (i == 61) {
143  ata->capacity |= ((uint32_t)data) << 16;
144  }
145  }
146 
147  return true;
148 }

150 int ata_write(device_t* device, uint32_t sectorNum, uint8_t* data, uint32_t size)
151 {
152  ata_t* dev = &_ata_drives[device->id];
153 
154  uint8_t dev_config = _ata_gen_drive_head_register(true, !dev->is_master, 0);
155 
156  port_8bit_out(dev->port.device, dev_config);
157  port_8bit_out(dev->port.sector_count, 1);
158  port_8bit_out(dev->port.lba_lo, sectorNum & 0x000000FF);
159  port_8bit_out(dev->port.lba_mid, (sectorNum & 0x0000FF00) >> 8);
160  port_8bit_out(dev->port.lba_hi, (sectorNum & 0x00FF0000) >> 16);
161  port_8bit_out(dev->port.error, 0);
162  port_8bit_out(dev->port.command, 0x31);
163 
164  // waiting for processing
165  // while BSY is on and no Errors
166  uint8_t status = port_8bit_in(dev->port.command);
167  while ((status >> 7) & 1 == 1 && (status >> 0) & 1 != 1) {
168  status = port_8bit_in(dev->port.command);
169  }
170 
171  // check if drive isn't ready to transer DRQ
172  if ((status >> 0) & 1 == 1) {
173  kprintf("Error");
174  return -EBUSY;
175  }
176 
177  for (int i = 0; i < size; i += 2) {
178  uint16_t db = (data[i + 1] << 8) + data[i];
179  port_16bit_out(dev->port.data, db);
180  }
181 
182  for (int i = size / 2; i < 256; i++) {
183  port_16bit_out(dev->port.data, 0);
184  }
185 
186  return ata_flush(device);
187 }

189 int ata_read(device_t* device, uint32_t sectorNum, uint8_t* read_data)
190 {
191  ata_t* dev = &_ata_drives[device->id];
192 
193  uint8_t dev_config = _ata_gen_drive_head_register(true, !dev->is_master, 0);
194 
195  port_8bit_out(dev->port.device, dev_config);
196  port_8bit_out(dev->port.sector_count, 1);
197  port_8bit_out(dev->port.lba_lo, sectorNum & 0x000000FF);
198  port_8bit_out(dev->port.lba_mid, (sectorNum & 0x0000FF00) >> 8);
199  port_8bit_out(dev->port.lba_hi, (sectorNum & 0x00FF0000) >> 16);
200  port_8bit_out(dev->port.error, 0);
201  port_8bit_out(dev->port.command, 0x21);
202 
203  // waiting for processing
204  // while BSY is on and no Errors
205  uint8_t status = port_8bit_in(dev->port.command);
206  while (((status >> 7) & 1) == 1 && ((status >> 0) & 1) != 1) {
207  status = port_8bit_in(dev->port.command);
208  }
209 
210  // check if drive isn't ready to transer DRQ
211  if (((status >> 0) & 1) == 1) {
212  kprintf("Error");
213  return -EBUSY;
214  }
215 
216  if (((status >> 3) & 1) == 0) {
217  kprintf("No DRQ");
218  return -ENODEV;
219  }
220 
221  for (int i = 0; i < 256; i++) {
222  uint16_t data = port_16bit_in(dev->port.data);
223  read_data[2 * i + 1] = (data >> 8) & 0xFF;
224  read_data[2 * i + 0] = (data >> 0) & 0xFF;
225  }
226 }

228 int ata_flush(device_t* device)
229 {
230  ata_t* dev = &_ata_drives[device->id];
231 
232  uint8_t dev_config = _ata_gen_drive_head_register(true, !dev->is_master, 0);
233 
234  port_8bit_out(dev->port.device, dev_config);
235  port_8bit_out(dev->port.command, 0xE7);
236 
237  uint8_t status = port_8bit_in(dev->port.command);
238  if (status == 0x00) {
239  return -ENODEV;
240  }
241 
242  while (((status >> 7) & 1) == 1 && ((status >> 0) & 1) != 1) {
243  status = port_8bit_in(dev->port.command);
244  }
245 
246  if (status & 0x01) {
247  return -EBUSY;
248  }
249 
250  return 0;
251 }
